# Phase 0b — App Scaffold Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Stand up an Expo SDK 54+ cross-platform skeleton for the Mauritian Rides app with all core libraries wired, route groups for guest/rider/driver personas, theme tokens, EN/FR i18n, MSW-mocked REST client, auth gate, and a dev client build that boots on iOS Simulator and Android Emulator.

**Architecture:** Single-codebase Expo app at `mobile-app/`, file-based routing via Expo Router v4, server state via TanStack Query v5 (currently against MSW mocks; swaps to live `mauritianrides.com/wp-json/mr/v1/` later), UI state via Zustand v5, styling via NativeWind v4 + gluestack-ui v2 unstyled primitives. New Architecture + bridgeless on by default (Expo SDK 54+). Backend integration (real JWT, MIPS, push, GPS) is deferred to subsequent plans; this plan ships a runnable shell with mocked data only.

**Tech Stack:** Expo SDK 54+, React Native 0.81+, Expo Router v4, TanStack Query v5, Zustand v5, NativeWind v4, gluestack-ui v2, MSW v2, Axios v1 + axios-retry v4, expo-secure-store, i18next v25 + react-i18next + expo-localization, Jest 30 + React Native Testing Library v13, TypeScript strict mode, EAS Build + Submit + Update.

```ts
import * as SecureStore from 'expo-secure-store';

// Access token lives only in memory. Cleared on app kill — refreshed silently
// on next launch from the persisted refresh token.
let accessToken: string | null = null;

export function getAccessToken(): string | null {
  return accessToken;
}

export function setAccessToken(token: string): void {
  accessToken = token;
}

export function clearAccessToken(): void {
  accessToken = null;
}

const REFRESH_KEY = 'mr_refresh';

export async function getRefreshToken(): Promise<string | null> {
  return SecureStore.getItemAsync(REFRESH_KEY);
}

export async function setRefreshToken(token: string): Promise<void> {
  await SecureStore.setItemAsync(REFRESH_KEY, token, {
    keychainAccessible: SecureStore.WHEN_UNLOCKED_THIS_DEVICE_ONLY,
  });
}

export async function clearRefreshToken(): Promise<void> {
  await SecureStore.deleteItemAsync(REFRESH_KEY);
}
```

```ts
  status: number;
  code: string;
  message: string;
}

  baseURL,
  timeout: 12_000,
  headers: { 'Content-Type': 'application/json' },
});

  const token = getAccessToken();
  if (token) {
    config.headers.Authorization = `Bearer ${token}`;
  }
  return config;

    const code = err.response?.data?.code ?? (err.code ?? 'network_error');
    const message = err.response?.data?.message ?? err.message;
    const apiError: ApiError = { status, code, message };
    return Promise.reject(apiError);
  },
);

axiosRetry(api, {
  retries: 3,
  retryDelay: axiosRetry.exponentialDelay,
  retryCondition: (e) => e.response?.status === 429,
```

```ts
import * as Sentry from '@sentry/react-native';
import Constants from 'expo-constants';

let initialized = false;

export function initSentry() {
  if (initialized) return;
  const dsn = Constants.expoConfig?.extra?.sentryDsn as string | undefined;
  if (!dsn) return; // skip in dev / when env var not set
  Sentry.init({
    dsn,
    enableAutoSessionTracking: true,
    tracesSampleRate: 0.2,
    environment: __DEV__ ? 'development' : 'production',
  });
  initialized = true;
}

export { Sentry };
```

```ts
import PostHog from 'posthog-react-native';
import Constants from 'expo-constants';

let client: PostHog | null = null;

export function initPostHog(): PostHog | null {
  if (client) return client;
  const key = Constants.expoConfig?.extra?.posthogKey as string | undefined;
  if (!key) return null;
  client = new PostHog(key, {
    host: 'https://eu.i.posthog.com',
    flushAt: 20,
    flushInterval: 30_000,
  });
  return client;
}

export function getPostHog(): PostHog | null {
  return client;
}
```
